class Solution:
    def threeSum(self, nums: list[int]) -> list[list[int]]:
        nums.sort()
        res = []
        n = len(nums)
        
        for i in range(n - 2): # Stop at n-2 because you need 3 numbers
            # 1. Skip duplicate 'i' values
            if i > 0 and nums[i] == nums[i-1]:
                continue
